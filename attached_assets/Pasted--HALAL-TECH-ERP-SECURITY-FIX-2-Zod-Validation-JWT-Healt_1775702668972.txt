```typescript
// helper function للتحقق
export function validate<T>(schema: z.ZodSchema<T>, data: unknown): 
  { success: true; data: T } | { success: false; errors: string[] } {
  const result = schema.safeParse(data);
  if (result.success) {
    return { success: true, data: result.data };
  }
  const errors = result.error.errors.map(e => `${e.path.join('.')}: ${e.message}`);
  return { success: false, errors };
}
```

## FIX 2: تحسين JWT — تقليل المدة + إضافة refresh

في `artifacts/api-server/src/middleware/auth.ts`:

```typescript
// تقليل مدة الـ access token من 12 ساعة إلى 4 ساعات
export function signToken(userId: number, role: string, companyId: number): string {
  return jwt.sign(
    { userId, role, companyId },
    JWT_SECRET,
    { expiresIn: '4h' }  // كان 12h
  );
}

// إضافة refresh token (مدة أطول - 7 أيام)
export function signRefreshToken(userId: number): string {
  const REFRESH_SECRET = process.env.JWT_REFRESH_SECRET || JWT_SECRET + '_refresh';
  return jwt.sign(
    { userId, type: 'refresh' },
    REFRESH_SECRET,
    { expiresIn: '7d' }
  );
}

export function verifyRefreshToken(token: string): { userId: number } | null {
  try {
    const REFRESH_SECRET = process.env.JWT_REFRESH_SECRET || JWT_SECRET + '_refresh';
    const decoded = jwt.verify(token, REFRESH_SECRET) as any;
    if (decoded.type !== 'refresh') return null;
    return { userId: decoded.userId };
  } catch {
    return null;
  }
}
```

في `artifacts/api-server/src/routes/auth.ts`:

```typescript
// عند تسجيل الدخول — أرجع access token + refresh token
const accessToken = signToken(user.id, user.role, user.company_id);
const refreshToken = signRefreshToken(user.id);

return res.json({
  token: accessToken,
  refreshToken,
  user: { ...userInfo }
});

// إضافة route جديد للـ refresh
router.post('/refresh', async (req, res) => {
  const { refreshToken } = req.body;
  if (!refreshToken) {
    return res.status(400).json({ error: 'refresh token مطلوب' });
  }
  
  const payload = verifyRefreshToken(refreshToken);
  if (!payload) {
    return res.status(401).json({ error: 'refresh token غير صالح أو منتهي' });
  }
  
  // جلب المستخدم من DB للتأكد إنه لا يزال نشطاً
  const user = await db.select().from(erpUsersTable)
    .where(and(eq(erpUsersTable.id, payload.userId), eq(erpUsersTable.active, true)))
    .limit(1);
  
  if (!user[0]) {
    return res.status(401).json({ error: 'المستخدم غير موجود أو موقوف' });
  }
  
  const newAccessToken = signToken(user[0].id, user[0].role, user[0].company_id);
  return res.json({ token: newAccessToken });
});
```

## FIX 3: Health Check Endpoint

أنشئ `artifacts/api-server/src/routes/health.ts`:

```typescript
import { Router } from 'express';
import { db } from '@workspace/db';
import { sql } from 'drizzle-orm';

const router = Router();

router.get('/', async (req, res) => {
  const start = Date.now();
  
  try {
    // اختبار الاتصال بقاعدة البيانات
    await db.execute(sql`SELECT 1`);
    const dbLatency = Date.now() - start;
    
    return res.json({
      status: 'healthy',
      timestamp: new Date().toISOString(),
      uptime: Math.floor(process.uptime()),
      database: {
        status: 'connected',
        latency_ms: dbLatency,
      },
      memory: {
        used_mb: Math.round(process.memoryUsage().heapUsed / 1024 / 1024),
        total_mb: Math.round(process.memoryUsage().heapTotal / 1024 / 1024),
      },
      version: process.env.npm_package_version || '1.0.0',
    });
  } catch (error) {
    return res.status(503).json({
      status: 'unhealthy',
      timestamp: new Date().toISOString(),
      database: { status: 'disconnected' },
      error: 'Database connection failed',
    });
  }
});

export default router;
```

## FIX 5: Global Error Handler محسّن

في `artifacts/api-server/src/app.ts`:
```typescript
// في نهاية الملف — بعد كل الـ routes
app.use((err: any, req: Request, res: Response, next: NextFunction) => {
  // لا ترسل stack trace للـ client أبداً
  const isDev = process.env.NODE_ENV !== 'production';
  
  // Zod validation errors
  if (err.name === 'ZodError') {
    return res.status(400).json({
      error: 'بيانات غير صحيحة',
      details: err.errors.map((e: any) => e.message),
    });
  }
  
  // JWT errors
  if (err.name === 'JsonWebTokenError' || err.name === 'TokenExpiredError') {
    return res.status(401).json({ error: 'جلسة منتهية، يرجى تسجيل الدخول مجدداً' });
  }
  
  // Database errors (لا تكشف تفاصيل DB)
  if (err.code && err.code.startsWith('23')) {
    return res.status(409).json({ error: 'البيانات موجودة مسبقاً أو تعارض في البيانات' });
  }
  
  // Generic error
  console.error('Unhandled error:', isDev ? err : err.message);
  return res.status(500).json({
    error: 'حدث خطأ في الخادم',
    ...(isDev && { details: err.message }),
  });
});
```
