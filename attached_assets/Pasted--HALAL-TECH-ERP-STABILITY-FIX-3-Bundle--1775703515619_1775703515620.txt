أضف الـ API endpoint لإرجاع معلومات الاشتراك:
```typescript
// في artifacts/api-server/src/routes/auth.ts
// GET /auth/subscription — يرجع معلومات اشتراك الشركة الحالية
router.get('/subscription', authenticate, async (req, res) => {
  if (req.role === 'super_admin') {
    return res.json({ unlimited: true });
  }
  
  const company = await db.select()
    .from(companiesTable)
    .where(eq(companiesTable.id, req.companyId))
    .limit(1);
  
  if (!company[0]) {
    return res.status(404).json({ error: 'الشركة غير موجودة' });
  }
  
  const today = new Date();
  const endDate = new Date(company[0].end_date);
  const daysLeft = Math.ceil((endDate.getTime() - today.getTime()) / (1000 * 60 * 60 * 24));
  
  return res.json({
    plan_type: company[0].plan_type,
    end_date: company[0].end_date,
    days_left: daysLeft,
    is_active: company[0].is_active,
    is_expiring_soon: daysLeft <= 14 && daysLeft > 0,
    is_expired: daysLeft <= 0,
  });
});
```

## FIX 7: Graceful Shutdown محسّن

في `artifacts/api-server/src/index.ts`:
```typescript
// تحسين الـ cleanup function
async function cleanup(signal: string) {
  logger.info({ signal }, 'Shutdown signal received');
  
  // أوقف استقبال requests جديدة
  server.close(async () => {
    try {
      // أغلق DB connections بشكل نظيف
      await pool.end();
      logger.info('Database pool closed');
    } catch (err) {
      logger.error({ err }, 'Error closing database pool');
    }
    logger.info('Server closed cleanly');
    process.exit(0);
  });
  
  // Force exit بعد 10 ثواني لو مش اتقفل
  setTimeout(() => {
    logger.error('Forced shutdown after timeout');
    process.exit(1);
  }, 10000);
}

process.on('SIGTERM', () => cleanup('SIGTERM'));
process.on('SIGINT', () => cleanup('SIGINT'));
```
