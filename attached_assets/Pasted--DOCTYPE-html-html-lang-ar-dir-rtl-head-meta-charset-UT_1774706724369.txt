<script>
    // تطبيق التغييرات على متغيرات CSS مباشرة
    const root = document.documentElement;

    function updatePrimaryColor(color) {
        root.style.setProperty('--primary-color', color);
        // تحسين الألوان المساعدة
        let r, g, b;
        if (color.startsWith('#')) {
            r = parseInt(color.slice(1,3), 16);
            g = parseInt(color.slice(3,5), 16);
            b = parseInt(color.slice(5,7), 16);
        } else {
            r = 44; g = 125; b = 160;
        }
        const primaryLight = `rgb(${Math.min(r+40, 255)}, ${Math.min(g+40, 255)}, ${Math.min(b+40, 255)})`;
        const primaryDark = `rgb(${Math.max(r-35, 20)}, ${Math.max(g-35, 20)}, ${Math.max(b-35, 20)})`;
        root.style.setProperty('--primary-light', primaryLight);
        root.style.setProperty('--primary-dark', primaryDark);
    }

    function updateFontFamily(font) {
        root.style.setProperty('--font-family', font);
    }

    function updateBorderWidth(width) {
        root.style.setProperty('--border-width', width + 'px');
        document.getElementById('borderWidthValue').innerText = width + 'px';
    }

    function updateFontWeight(weight) {
        root.style.setProperty('--font-weight-normal', weight);
        let boldWeight = parseInt(weight) + 100;
        if (boldWeight > 800) boldWeight = 800;
        root.style.setProperty('--font-weight-bold', boldWeight);
    }

    function updateIconSize(size) {
        root.style.setProperty('--icon-size', size + 'px');
        document.getElementById('iconSizeValue').innerText = size + 'px';
    }

    // ربط العناصر
    const colorPicker = document.getElementById('primaryColor');
    const fontSelect = document.getElementById('fontFamily');
    const borderSlider = document.getElementById('borderWidth');
    const fontWeightSelect = document.getElementById('fontWeight');
    const iconSizeSlider = document.getElementById('iconSize');

    colorPicker.addEventListener('input', (e) => updatePrimaryColor(e.target.value));
    fontSelect.addEventListener('change', (e) => updateFontFamily(e.target.value));
    borderSlider.addEventListener('input', (e) => updateBorderWidth(parseFloat(e.target.value)));
    fontWeightSelect.addEventListener('change', (e) => updateFontWeight(e.target.value));
    iconSizeSlider.addEventListener('input', (e) => updateIconSize(parseInt(e.target.value)));

    // حفظ الإعدادات في localStorage (اختياري)
    document.getElementById('applySettings').addEventListener('click', () => {
        const settings = {
            primaryColor: colorPicker.value,
            fontFamily: fontSelect.value,
            borderWidth: borderSlider.value,
            fontWeight: fontWeightSelect.value,
            iconSize: iconSizeSlider.value
        };
        localStorage.setItem('appSettings', JSON.stringify(settings));
        alert('تم حفظ الإعدادات! سيتم تطبيقها في下次 فتح التطبيق.');
    });

    // استرجاع الإعدادات المحفوظة عند تحميل الصفحة
    function loadSettings() {
        const saved = localStorage.getItem('appSettings');
        if (saved) {
            try {
                const settings = JSON.parse(saved);
                colorPicker.value = settings.primaryColor;
                updatePrimaryColor(settings.primaryColor);
                fontSelect.value = settings.fontFamily;
                updateFontFamily(settings.fontFamily);
                borderSlider.value = settings.borderWidth;
                updateBorderWidth(parseFloat(settings.borderWidth));
                fontWeightSelect.value = settings.fontWeight;
                updateFontWeight(settings.fontWeight);
                iconSizeSlider.value = settings.iconSize;
                updateIconSize(parseInt(settings.iconSize));
            } catch(e) {}
        }
    }
    loadSettings();
</script>
