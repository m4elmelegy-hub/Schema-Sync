### إنشاء 2FA Service
أنشئ `artifacts/api-server/src/lib/totp.ts`:

```typescript
import speakeasy from 'speakeasy';
import QRCode from 'qrcode';

const APP_NAME = 'Halal Tech ERP';

export function generateTOTPSecret(username: string) {
  const secret = speakeasy.generateSecret({
    name: `${APP_NAME} (${username})`,
    issuer: APP_NAME,
    length: 32,
  });
  return {
    secret: secret.base32,
    otpauth_url: secret.otpauth_url!,
  };
}

export async function generateQRCode(otpauth_url: string): Promise<string> {
  return await QRCode.toDataURL(otpauth_url);
}

export function verifyTOTP(secret: string, token: string): boolean {
  return speakeasy.totp.verify({
    secret,
    encoding: 'base32',
    token,
    window: 2, // يقبل ±2 دقيقة
  });
}
```

### إضافة API endpoints للـ 2FA
في `artifacts/api-server/src/routes/auth.ts`:

```typescript
import { generateTOTPSecret, generateQRCode, verifyTOTP } from '../lib/totp';

// GET /api/auth/2fa/setup — يبدأ إعداد الـ 2FA (super_admin فقط)
router.get('/2fa/setup', authenticate, async (req, res) => {
  if (req.role !== 'super_admin') {
    return res.status(403).json({ error: 'للمسؤول العام فقط' });
  }

  const user = await db.select().from(erpUsersTable)
    .where(eq(erpUsersTable.id, req.userId)).limit(1);

  if (user[0]?.totp_enabled) {
    return res.status(400).json({ error: '2FA مفعل بالفعل' });
  }

  const { secret, otpauth_url } = generateTOTPSecret(user[0].username);
  const qrCode = await generateQRCode(otpauth_url);

  // احفظ الـ secret مؤقتاً (لم يُفعَّل بعد)
  await db.update(erpUsersTable)
    .set({ totp_secret: secret, totp_verified: false })
    .where(eq(erpUsersTable.id, req.userId));

  return res.json({
    qr_code: qrCode,           // صورة QR Code بـ base64
    secret,                     // للإدخال اليدوي
    message: 'امسح الـ QR Code بتطبيق Google Authenticator أو Authy',
  });
});

// POST /api/auth/2fa/verify — تأكيد تفعيل الـ 2FA
router.post('/2fa/verify', authenticate, async (req, res) => {
  if (req.role !== 'super_admin') {
    return res.status(403).json({ error: 'للمسؤول العام فقط' });
  }

  const { token } = req.body;
  if (!token) return res.status(400).json({ error: 'رمز التحقق مطلوب' });

  const user = await db.select().from(erpUsersTable)
    .where(eq(erpUsersTable.id, req.userId)).limit(1);

  if (!user[0]?.totp_secret) {
    return res.status(400).json({ error: 'يجب إعداد 2FA أولاً' });
  }

  const isValid = verifyTOTP(user[0].totp_secret, token);
  if (!isValid) {
    return res.status(400).json({ error: 'رمز التحقق غير صحيح' });
  }

  // فعّل الـ 2FA
  await db.update(erpUsersTable)
    .set({ totp_enabled: true, totp_verified: true })
    .where(eq(erpUsersTable.id, req.userId));

  return res.json({ success: true, message: 'تم تفعيل المصادقة الثنائية بنجاح ✅' });
});

// POST /api/auth/2fa/disable — إيقاف الـ 2FA
router.post('/2fa/disable', authenticate, async (req, res) => {
  if (req.role !== 'super_admin') {
    return res.status(403).json({ error: 'للمسؤول العام فقط' });
  }

  const { token } = req.body;
  const user = await db.select().from(erpUsersTable)
    .where(eq(erpUsersTable.id, req.userId)).limit(1);

  if (!user[0]?.totp_enabled) {
    return res.status(400).json({ error: '2FA غير مفعل' });
  }

  // تحقق من الـ token قبل الإيقاف
  const isValid = verifyTOTP(user[0].totp_secret!, token);
  if (!isValid) {
    return res.status(400).json({ error: 'رمز التحقق غير صحيح' });
  }

  await db.update(erpUsersTable)
    .set({ totp_enabled: false, totp_secret: null, totp_verified: false })
    .where(eq(erpUsersTable.id, req.userId));

  return res.json({ success: true, message: 'تم إيقاف المصادقة الثنائية' });
});
```

## FIX 3: IP Allowlist للـ Super Admin

في `artifacts/api-server/src/middleware/auth.ts`:

```typescript
// IP Allowlist للـ super_admin (اختياري — إذا كان SUPER_ADMIN_IPS مضبوطاً)
export function superAdminIPGuard(req: Request, res: Response, next: NextFunction) {
  const allowedIPs = process.env.SUPER_ADMIN_IPS?.split(',').map(ip => ip.trim());
  
  // لو مفيش IPs محددة — اسمح للكل
  if (!allowedIPs || allowedIPs.length === 0) {
    return next();
  }
  
  const clientIP = req.ip || 
    req.headers['x-forwarded-for']?.toString().split(',')[0].trim() || 
    req.socket.remoteAddress;
  
  if (!clientIP || !allowedIPs.includes(clientIP)) {
    return res.status(403).json({ 
      error: 'الوصول مرفوض — عنوان IP غير مصرح به',
    });
  }
  
  next();
}
```

## FIX 4: Session Blacklist — إلغاء الـ tokens

في `artifacts/api-server/src/lib/session-blacklist.ts`:

```typescript
// In-memory blacklist (يُعاد تعيينه عند restart — كافٍ للـ MVP)
const blacklistedTokens = new Set<string>();

export function blacklistToken(token: string): void {
  blacklistedTokens.add(token);
  // نظّف التوكنات المنتهية كل ساعة
  setTimeout(() => blacklistedTokens.delete(token), 4 * 60 * 60 * 1000);
}

export function isTokenBlacklisted(token: string): boolean {
  return blacklistedTokens.has(token);
}
```
