```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest';
import request from 'supertest';
import app from '../app';
import { signToken, signRefreshToken, verifyRefreshToken } from '../middleware/auth';
import { verifyTOTP } from '../lib/totp';

describe('Auth Middleware', () => {
  describe('signToken', () => {
    it('يجب أن ينشئ JWT صالح', () => {
      const token = signToken(1, 'company_admin', 1);
      expect(token).toBeDefined();
      expect(typeof token).toBe('string');
      expect(token.split('.')).toHaveLength(3);
    });

    it('يجب أن يحتوي على userId و role و companyId', () => {
      const jwt = require('jsonwebtoken');
      const token = signToken(42, 'cashier', 5);
      const decoded = jwt.verify(token, process.env.JWT_SECRET!) as any;
      expect(decoded.userId).toBe(42);
      expect(decoded.role).toBe('cashier');
      expect(decoded.companyId).toBe(5);
    });

    it('يجب أن ينتهي بعد 4 ساعات', () => {
      const jwt = require('jsonwebtoken');
      const token = signToken(1, 'admin', 1);
      const decoded = jwt.verify(token, process.env.JWT_SECRET!) as any;
      const expiresIn = decoded.exp - decoded.iat;
      expect(expiresIn).toBe(4 * 60 * 60);
    });
  });

  describe('signRefreshToken', () => {
    it('يجب أن ينشئ refresh token صالح', () => {
      const token = signRefreshToken(1);
      expect(token).toBeDefined();
    });

    it('يجب أن يتحقق من refresh token صحيح', () => {
      const token = signRefreshToken(1);
      const result = verifyRefreshToken(token);
      expect(result).not.toBeNull();
      expect(result?.userId).toBe(1);
    });

    it('يجب أن يرفض refresh token منتهي أو غلط', () => {
      const result = verifyRefreshToken('invalid-token');
      expect(result).toBeNull();
    });
  });
});

describe('POST /api/auth/login', () => {
  beforeEach(() => {
    const { db } = require('@workspace/db');
    // mock مستخدم نشط
    db.select.mockReturnValue({
      from: vi.fn().mockReturnValue({
        where: vi.fn().mockReturnValue({
          limit: vi.fn().mockResolvedValue([{
            id: 1,
            username: 'testuser',
            name: 'Test User',
            pin: '$2b$10$92IXUNpkjO0rOQ5byMi.Ye4oKoEa3Ro9llC/.og/at2.uHlHWY2W2', // "password"
            role: 'company_admin',
            active: true,
            company_id: 1,
            totp_enabled: false,
            login_attempts: 0,
          }])
        })
      })
    });
  });

  it('يجب أن يرفض login بدون userId', async () => {
    const res = await request(app)
      .post('/api/auth/login')
      .send({ pin: '123456' });
    expect(res.status).toBe(400);
  });

  it('يجب أن يرفض login بدون pin', async () => {
    const res = await request(app)
      .post('/api/auth/login')
      .send({ userId: 1 });
    expect(res.status).toBe(400);
  });

  it('يجب أن يرجع 400 لـ userId غير صالح', async () => {
    const res = await request(app)
      .post('/api/auth/login')
      .send({ userId: 'not-a-number', pin: '123456' });
    expect(res.status).toBe(400);
  });
});

describe('GET /api/healthz', () => {
```

## TESTS 2: Security Tests

أنشئ `artifacts/api-server/src/__tests__/security.test.ts`:
```typescript
import { describe, it, expect, vi } from 'vitest';
import request from 'supertest';
import app from '../app';
import { isTokenBlacklisted, blacklistToken } from '../lib/session-blacklist';
import { verifyTOTP, generateTOTPSecret } from '../lib/totp';
import speakeasy from 'speakeasy';

describe('Session Blacklist', () => {
  it('يجب أن يضيف token للـ blacklist', () => {
    const token = 'test-token-123';
    blacklistToken(token);
    expect(isTokenBlacklisted(token)).toBe(true);
  });

  it('يجب أن يقبل token غير موجود في الـ blacklist', () => {
    expect(isTokenBlacklisted('non-blacklisted-token')).toBe(false);
  });
});

describe('TOTP', () => {
  it('يجب أن ينشئ secret صالح', () => {
    const { secret, otpauth_url } = generateTOTPSecret('testuser');
    expect(secret).toBeDefined();
    expect(secret.length).toBeGreaterThan(10);
    expect(otpauth_url).toContain('testuser');
  });

  it('يجب أن يتحقق من TOTP code صحيح', () => {
    const secret = speakeasy.generateSecret({ length: 20 });
    const token = speakeasy.totp({ secret: secret.base32, encoding: 'base32' });
    const isValid = verifyTOTP(secret.base32, token);
    expect(isValid).toBe(true);
  });

  it('يجب أن يرفض TOTP code خاطئ', () => {
    const { secret } = generateTOTPSecret('testuser');
    const isValid = verifyTOTP(secret, '000000');
    expect(isValid).toBe(false);
  });
});

describe('Security Headers', () => {
  it('يجب أن يحتوي على X-Content-Type-Options', async () => {
    const res = await request(app).get('/api/healthz');
    expect(res.headers['x-content-type-options']).toBe('nosniff');
  });

  it('يجب أن يحتوي على X-Frame-Options', async () => {
    const res = await request(app).get('/api/healthz');
    expect(res.headers['x-frame-options']).toBeDefined();
  });

  it('يجب أن لا يكشف X-Powered-By', async () => {
    const res = await request(app).get('/api/healthz');
    expect(res.headers['x-powered-by']).toBeUndefined();
  });
});

describe('Rate Limiting', () => {
  it('يجب أن يحظر بعد تجاوز الـ rate limit', async () => {
    // إرسال 15 طلب متتالي على auth endpoint
    const requests = Array(15).fill(null).map(() =>
      request(app).post('/api/auth/login').send({ userId: 1, pin: 'wrong' })
    );
    const responses = await Promise.all(requests);
    const blockedResponses = responses.filter(r => r.status === 429);
    expect(blockedResponses.length).toBeGreaterThan(0);
  });
});

describe('Input Sanitization', () => {
  it('يجب أن ينظف XSS في الـ body', async () => {
    const res = await request(app)
      .post('/api/auth/login')
      .send({ 
        userId: 1, 
        pin: '<script>alert("xss")</script>' 
      });
    // يجب أن يمر الـ request (لكن الـ pin منظف)
    expect(res.status).not.toBe(500);
  });
});
```

```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest';
import request from 'supertest';
import app from '../app';
import { signToken } from '../middleware/auth';

describe('Multi-Tenant Isolation', () => {
  const company1Token = signToken(1, 'company_admin', 1);
  const company2Token = signToken(2, 'company_admin', 2);
  const superAdminToken = signToken(99, 'super_admin', 1);

  beforeEach(() => {
    const { db } = require('@workspace/db');
    
    // mock المستخدم مع company_id
    db.select.mockReturnValue({
      from: vi.fn().mockReturnValue({
        where: vi.fn().mockReturnValue({
          limit: vi.fn().mockResolvedValue([{
            id: 1,
            role: 'company_admin',
            active: true,
            company_id: 1,
          }])
        }),
        mockResolvedValue: vi.fn().mockResolvedValue([])
      })
    });
  });

  it('يجب أن يرفض الطلب بدون token', async () => {
    const res = await request(app).get('/api/settings/users');
    expect(res.status).toBe(401);
  });

  it('يجب أن يقبل الطلب بـ token صالح', async () => {
    const res = await request(app)
      .get('/api/settings/users')
      .set('Authorization', `Bearer ${company1Token}`);
    expect(res.status).not.toBe(401);
    expect(res.status).not.toBe(403);
  });

  it('يجب أن يرفض الوصول لـ super routes من company_admin', async () => {
    const res = await request(app)
      .get('/api/super/companies')
      .set('Authorization', `Bearer ${company1Token}`);
    expect(res.status).toBe(403);
  });

  it('يجب أن يسمح للـ super_admin بالوصول لـ super routes', async () => {
    const { db } = require('@workspace/db');
    db.select.mockReturnValue({
      from: vi.fn().mockReturnValue({
        where: vi.fn().mockReturnValue({
          limit: vi.fn().mockResolvedValue([{
            id: 99,
            role: 'super_admin',
            active: true,
            company_id: 1,
          }])
        }),
        mockResolvedValue: vi.fn().mockResolvedValue([])
      })
    });

    const res = await request(app)
      .get('/api/super/companies')
      .set('Authorization', `Bearer ${superAdminToken}`);
    expect(res.status).not.toBe(403);
  });
});
```
